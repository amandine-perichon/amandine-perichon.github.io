Three octopuses in an aquarium

This is the story of three octopuses in an aquarium
- or what 'scope' is, and how it works in JavaScript!


"You should imagine variables as tentacles, rather than boxes. They do not contain values; they grasp them"
Eloquent Javascript. Marijn Haverbeke 



// This is the story of bigOctopus, smallOctopus, and tinyOctopus 
// and their toy sharing arrangement inside bigAquarium

// bigOctopus lives in the bigAquarium
// In the bigAquarium, you can find two smaller aquariums: IF aquarium and smallAquarium
function bigAquarium () {
  // bigOctopus has a big heart! He always lets everyone play with his toys
  // That includes the other octopus in the bigAquarium: smallOctopus and tinyOctopus
  var bigOctopusTentacleOne = "ja";
  var bigOctopusTentacleThree = "va";
  var bigOctopusTentacleFour;
  var bigOctopusTentacleFive;
  // Somehow, bigOctopus's tentacle eight got stuck in a toy
  // He is condamned to play with toy "pt" for the rest of his life!
  const bigOctopusTentacleEight = "pt";

  // tinyOctopus lives in the IF aquarium
  // tinyOctopus does not let anyone outside of his tiny IF aquarium play with his toys
  // but it sure can play with the toys from the BigAquarium
  if (true) {
    // tinyOctopusTentacleTwo's toy "sc" can only be played with in the IF aquarium
    let tinyOctopusTentacleTwo = "sc";
    // bigOctopusTentacleFour's toy can be played with in the IF aquarium
    bigOctopusTentacleFour = tinyOctopusTentacleTwo;
  }

  // smallOctopus lives in the smallAquarium
  // smallOctopus does not let anyone outside of his smallAquarium play with his toys
  // but it sure can play with the toys from the BigAquarium
  function smallAquarium () {
    // smallOctopusTentacleTwo's toy "ri" can only be played with in the smallAquarium
    var smallOctopusTentacleFour = "ri";
    // bigOctopusTentacleFive's toy can be played with in smallAquarium
    bigOctopusTentacleFive = smallOctopusTentacleFour;
  }

  return bigOctopusTentacleOne + bigOctopusTentacleThree + bigOctopusTentacleFour 
          + bigOctopusTentacleFive + bigOctopusTentacleEight 
}

console.log("Scope of a toy/variable = Aquariums/functions/blocks in which it can be played with")
console.log( "This is how you handle scope in " + bigAquarium());
